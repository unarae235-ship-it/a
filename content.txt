  <script>
    const lines = [
      "오늘 너의 실행 속도: 광속. 아직 시작 안 했는데도 벌써 끝났어.",
      "너의 아이디어, 냉장고 열 때 불 켜지는 수준으로 자동 명작.",
      "지구가 둥근 이유: 네가 중심이니까.",
      "코드 안 써도 버그가 겁먹고 도망갈 포스.",
      "네 웃음소리 듣고 Wi-Fi가 더 빨라졌대.",
      "오늘의 운세: 칭찬 과다복용 주의 🥰",
      "네가 말하면 세미콜론도 스스로 붙는다 ;",
      "네 존재감, 다크모드보다 눈에 편해.",
      "네 앞으로 펼칠 일들, 캐시 미리 적중 완료.",
      "방금 지나간 행운… 다시 돌아왔어. 네가 있어서."
    ];

    const out = document.getElementById("out");
    const year = document.getElementById("year");
    year.textContent = new Date().getFullYear();

    function pick() {
      out.style.opacity = 0.3;
      setTimeout(() => {
        const i = Math.floor(Math.random() * lines.length);
        out.textContent = lines[i];
        out.style.opacity = 1;
      }, 120);
    }

    document.getElementById("go").addEventListener("click", pick);

    document.getElementById("copy").addEventListener("click", async () => {
      try {
        await navigator.clipboard.writeText(out.textContent.trim());
        alert("복사 완료!");
      } catch {
        alert("복사를 지원하지 않는 브라우저예요 😅");
      }
    });

    document.getElementById("share").addEventListener("click", async () => {
      const text = out.textContent.trim();
      const shareData = { title: "랜덤 칭찬", text, url: location.href };
      if (navigator.share) {
        try { await navigator.share(shareData); }
        catch(e) { }
      } else {
        try {
          await navigator.clipboard.writeText(text + " " + location.href);
          alert("링크와 함께 복사했어요!");
        } catch {
          alert("공유 미지원: 복사 버튼을 사용해 주세요.");
        }
      }
    });

    pick();
  </script>
